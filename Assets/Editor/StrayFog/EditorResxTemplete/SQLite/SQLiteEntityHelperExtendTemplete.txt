using System.Collections.Generic;
/// <summary>
/// SQLite表实体帮助类
/// </summary>
public sealed partial class SQLiteEntityHelper
{
	/// <summary>
    /// 实体设定映射
    /// </summary>
    static readonly Dictionary<int, SQLiteEntitySetting> msrEntityMaping = new Dictionary<int, SQLiteEntitySetting>() {#EntityMaping#
		{ typeof(#ClassName#).GetHashCode(),new SQLiteEntitySetting(typeof(#ClassName#).GetHashCode(),"#TableName#","#XlsFileName#",#IsDeterminant#,enSQLiteEntityClassify.#Classify#,#xlsColumnNameIndex#,#xlsColumnDataIndex#)},#EntityMaping#
    };

    /// <summary>
    /// 获得实体设定
    /// </summary>
    /// <returns>实体设定</returns>
    static SQLiteEntitySetting OnGetEntitySetting<T>()
        where T : AbsSQLiteEntity
    {
        int key = typeof(T).GetHashCode();
        return msrEntityMaping[key];
    }
}
