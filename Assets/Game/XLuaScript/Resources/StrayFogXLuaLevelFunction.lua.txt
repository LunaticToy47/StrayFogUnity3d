local selfTable = {}
local speed = 10;
local lightCpnt = nil;
local cube = nil;
local x = 1;
local y = 2;

function selfTable.setcube(c)
	cube = c;
end

function selfTable.awake()
	print("lua awake...");
	print("/",x/y);
end

function selfTable.start()
	print("lua start...");
	lightCpnt= CS.UnityEngine.GameObject.FindObjectOfType(typeof(CS.UnityEngine.Light));
end

function selfTable.update()
	print("lua update...");
	if(lightCpnt ~= nil)
	then
		local r = CS.UnityEngine.Vector3.up * CS.UnityEngine.Time.deltaTime * speed;
		print(cube);
		cube.transform:Rotate(r);
		lightCpnt.color = CS.UnityEngine.Color(CS.UnityEngine.Mathf.Sin(CS.UnityEngine.Time.time) / 2 + 0.5, 0, 0, 1);
	end
end

function selfTable.ondestroy()
    print("lua destroy");
end

return selfTable
